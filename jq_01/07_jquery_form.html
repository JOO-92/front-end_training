val()

val() 메소드는 입력 요소에 있는 value 속성값을 불러오거나 변경할 때 사용한다. val() 메소드는 value 속성값을 반환할 때와 value 속성값을 바꿀 때 사용하는 형식이 다르다.

function displayVals() {
  var singleValues = $( "#single" ).val();
  var multipleValues = $( "#multiple" ).val();
  $(".result").html( "<b>Single:</b> " + singleValues +
    " <b>Multiple:</b>" + multipleValues.join(", "));
}
$(document).ready(function(){
  $("select").change( displayVals );
  displayVals();
});

  $( ".result" ).text(value);
});

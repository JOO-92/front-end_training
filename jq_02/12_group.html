그룹 이벤트 등록 메소드

그룹 이벤트 등록 메소드를 사용하면 한 번에 2개 이상의 이벤트를 등록할 수 있다. 즉 선택한 요소에 메소드를 한 번만 적용하여 두 이벤트 이상을 등록할 수 있는 것이다.

on()

on() 이벤트는 1개 이상의 이벤트에 대해 동일한 동작을 지정할 수도 있고, 여러 개의 이벤트에 각각의 동작을 지정할 수도 있다.

$(document).ready(function(){
  // 두가지 모두 가능한 형태이다.
  $('input').on({
    'click':function(){
      alert('왜 눌러!')
    },
    'keyup':function(){
      const value = $(this).val();
      console.log(value);
    },
    'mouseover':function(){
      $(this).css('backgroundColor', 'yellow')
    }
  })
  $('input').on('click keyup mouseover', function(){
    console.log('!');
  })
});

delegate()

delegate()는 선택한 요소의 하위 요소에 여러 개의 이벤트를 등록할 수 있다. 기본 형태는 다음과 같다.

3
4
5
$(document).ready(function(){
  $(상위요소선택).delegate('하위 요소 선택', '이벤트 종류', function(){
    
  })
});

$(document).ready(function(){
  $('form').delegate('input', 'click', function(){
    $(this).css('backgroundColor', 'red');
  })
});

one()

선택한 요소에 이벤트를 등록하되, 한번 이벤트가 발생되면 자동으로 등록된 이벤트가 제거된다.

1
2
3
4
5
$(document).ready(function(){
  $('input').one('click', function(){
    console.log('!')
  })
});

이벤트 제거 메소드 off()

off()를 사용해 등록되어있던 이벤트를 제거할 수 있다.

$(document).ready(function(){
  $('section > button').on('mouseover', function(){
    $(this).css("backgroundColor", "yellow");
  })
  $('section > button').on('click', function(){
    $(this).css("backgroundColor", "red");
    $(this).off();
  })
});

효과 메소드

효과 메소드를 사용하면 스타일(CSS)를 사용하는 것보다 좀 더 역동적인 동작을 적용하여 객체를 화려하게 숨기거나 보이게 할 수 있다. 효과 메소드에는 CSS 속성을 바꾸는 메소드와 애니메이션 기능을 다루는 메소드가 있다.

hide() / show()

요소를 숨기거나 숨겨져 있는 요소를 노출시키는 메소드들이다.

1
2
$(선택요소).show(속도_ms, 함수_생략가능);
$(선택요소).hide(속도_ms, 함수_생략가능);

예제

1
2
3
4
5
6
7
8

$(document).ready(function(){
  let count = 0;
 
  $("#lab > button").click(function(){
    if(count % 2 == 0){
      $(this).prev().children().first().hide()
      count++;
    }else{
      $(this).prev().children().first().show()
      count++;
    }
  })
})

    if(count % 2 == 0){
      $(this).prev().children().first().toggle()
      count++;
    }else{
      $(this).prev().children().first().toggle()
      count++;
    }
  })
})

fadeIn() / fadeOut() / fadeToggle()

나타나고 사라진다는 기본 사항은 모두 비슷한데, fade 메소드는 점점 투명해지거나 점점 흐려지는 효과까지 포함한다. hide(), show()와 동일한 매개변수 사용이 가능하다.

1
2
3
4
5
$(document).ready(function(){
  $("#lab > button").click(function(){
    $('img').fadeToggle();
  })
});

fadeTo()

fade 효과를 지정한 투명도만큼만 주게 된다.

1
$('선택요소').fadeTo(속도, 투명도, 함수);

예제

1
2
3
4

$(document).ready(function(){
  let count = 0;
 
  $("#lab > button").click(function(){
    if(count % 2 == 0){
      $(this).prev().fadeTo(1000, 0.5);
      count++;
    }else{
      $(this).prev().fadeTo(0, 1, function(){window.alert("원상복구!")});
      count++;
    }
  })
})

animate()

선택한 요소에 다양한 모션을 적용할 수 있다. 애니메이션 효과. animate() 메소드는 연달아 지정하는 것이 가능하다.

1
$('선택요소').animate({애니메이션속성},속도,함수);

예제

1
2
3
4
5
6

8

$(document).ready(function(){
  $("#move").animate({marginLeft:100}, 1000, function(){
    $('#result').append(document.createElement('p').innerText = "1번 끝 ");
  }).animate({marginTop:100}, 1000, function(){
    $('#result').append(document.createElement('p').innerText = "2번 끝 ");
  }).animate({marginLeft:0}, 1000, function(){
    $('#result').append(document.createElement('p').innerText = "3번 끝 ");
  }).animate({marginTop:0}, 1000, function(){
    $('#result').append(document.createElement('p').innerText = "4번 끝 ");
  })
})

$(document).ready(function(){
  var mySlider = $('#slide_banner').bxSlider({
    speed : 500, // 이동 속도 정하기(ms)
    pager : false, // 현재 페이지 표시 안 하겠다!
    moveSlides : 2, // 이동할 때 몇 장씩 이동할지..
    slideWidth : 90, // 한 슬라이드 당 너비.
    maxSlides : 4, // 최대 몇 장까지 보여줄지
    slideMargin : 10, // 슬라이드 간 간격
    controls : false // 버튼 노출 여부
  })
  
  // 버튼 동작 등록
  $('#prevBtn').on('click', function(){
    // 내 슬라이더에 이동하는 동작을 명령하면 됨.
    mySlider.goToPrevSlide()
    return false
  })
  $('#nextBtn').on('click', function(){
    // 내 슬라이더에 이동하는 동작을 명령하면 됨.
    mySlider.goToNextSlide()
    return false
  })
});

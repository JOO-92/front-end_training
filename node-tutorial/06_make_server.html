http 모듈로 서버 만들기

요청과 응답

서버와 클라이언트는 상호작용 한다. 클라이언트에서 서버로 요청을 보내고, 서버에서는 요청의 내용을 읽고 처리한 뒤 클라이언트에게 응답을 보낸다. 따라서 서버에는 요청을 받는 부분과 응답을 보내는 부분이 있어야 한다. 요청과 응답은 각각 이벤트로써 처리하는 방식을 사용할 것이다. 클라이언트로부터 어떤 요청이 들어왔을 때 어떤 작업을 수행할지 이벤트 리스너를 미리 등록해두어야 한다.

아래는 채소 목록을 응답으로 전달해주는 서버의 예제이다.

const http = require('http');
// http 모듈을 사용할 것이다.
 
http.createServer((req, res) => {
  res.writeHead(200, {"Content-Type": "text/html; charset=utf-8"});
  res.write(`
    <h1>맛있는 채소</h1>
    <ul>
      <li>양배추</li>
      <li>오이</li>
      <li>적양파</li>
    </ul>
  `, "utf-8");
  res.end();
}).listen(8080, () => {
  console.log("서버가 응답을 했습니다!")
})

listen 메소드를 사용하는 방법 외에도 listening 이벤트에 대한 이벤트 리스너를 등록하는 방법도 가능하다.

const http = require('http');
// http 모듈을 사용할 것이다.
 
const server = http.createServer((req, res) => {
  res.writeHead(200, {"Content-Type": "text/html; charset=utf-8"});
  res.write(`
    <h1>맛있는 채소</h1>
    <ul>
      <li>양배추</li>
      <li>오이</li>
      <li>적양파</li>
    </ul>
  `, "utf-8");
  res.end();
})
 
server.listen(8080);
server.on('listening', () => {
  console.log("서버가 응답을 했습니다!")
})
// 에러 상황에 대응하는 이벤트 리스너 추가!
server.on('error', (err) => {
  console.error(err);
})

const http = require('http');
const fs = require('fs');
 
http.createServer((req, res) => {
  fs.readFile('./07_index.html', (err, data) => {
    if(err){
      console.error(err);
    }
    res.end(data);
  })
}).listen(8080, () => {
  console.log("서버가 응답을 했습니다!")
})
